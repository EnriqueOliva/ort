## 🎯 **1. TEST-TIME AUGMENTATION (TTA)**

TTA improves segmentation accuracy without retraining by augmenting test images, predicting masks, inverting transformations, and aggregating predictions. Scientific research on cell segmentation demonstrated TTA's effectiveness with both Mask R-CNN and U-Net architectures.

### **Best Augmentations for U-Net Segmentation:**

```python
import torch
import torch.nn.functional as F

class TTAWrapper:
    def __init__(self, model):
        self.model = model
        self.model.eval()
    
    def predict_with_tta(self, image):
        """
        Apply TTA with horizontal flip, vertical flip, and 90° rotations
        """
        predictions = []
        
        # Original
        with torch.no_grad():
            pred = torch.sigmoid(self.model(image))
            predictions.append(pred)
        
        # Horizontal flip
        with torch.no_grad():
            flipped = torch.flip(image, dims=[3])
            pred = torch.sigmoid(self.model(flipped))
            pred = torch.flip(pred, dims=[3])
            predictions.append(pred)
        
        # Vertical flip
        with torch.no_grad():
            flipped = torch.flip(image, dims=[2])
            pred = torch.sigmoid(self.model(flipped))
            pred = torch.flip(pred, dims=[2])
            predictions.append(pred)
        
        # 90° rotation
        with torch.no_grad():
            rotated = torch.rot90(image, k=1, dims=[2, 3])
            pred = torch.sigmoid(self.model(rotated))
            pred = torch.rot90(pred, k=-1, dims=[2, 3])
            predictions.append(pred)
        
        # 180° rotation
        with torch.no_grad():
            rotated = torch.rot90(image, k=2, dims=[2, 3])
            pred = torch.sigmoid(self.model(rotated))
            pred = torch.rot90(pred, k=-2, dims=[2, 3])
            predictions.append(pred)
        
        # 270° rotation
        with torch.no_grad():
            rotated = torch.rot90(image, k=3, dims=[2, 3])
            pred = torch.sigmoid(self.model(rotated))
            pred = torch.rot90(pred, k=-3, dims=[2, 3])
            predictions.append(pred)
        
        # Average all predictions
        final_pred = torch.stack(predictions).mean(dim=0)
        return final_pred

# Usage
tta_model = TTAWrapper(model)
enhanced_prediction = tta_model.predict_with_tta(test_image)
```

## 🧹 **2. MORPHOLOGICAL POST-PROCESSING**

Morphological operations are crucial for smoothing outlines, removing single pixels, and filling black holes in white regions.

```python
from skimage.morphology import (
    binary_opening, binary_closing, 
    remove_small_objects, remove_small_holes,
    disk, binary_dilation, binary_erosion
)
from scipy.ndimage import binary_fill_holes
import numpy as np

def morphological_postprocess(mask, min_object_size=500, min_hole_size=500):
    """
    Apply morphological operations to clean up binary mask
    
    Args:
        mask: Binary mask (H, W) with values 0 or 1
        min_object_size: Minimum size of objects to keep (in pixels)
        min_hole_size: Minimum size of holes to fill (in pixels)
    """
    # Convert to boolean
    mask_bool = mask.astype(bool)
    
    # Step 1: Remove small objects (noise)
    mask_clean = remove_small_objects(mask_bool, min_size=min_object_size)
    
    # Step 2: Fill holes
    mask_filled = binary_fill_holes(mask_clean)
    
    # Step 3: Remove small holes
    mask_filled = remove_small_holes(mask_filled, area_threshold=min_hole_size)
    
    # Step 4: Opening (erosion + dilation) to smooth boundaries
    selem = disk(3)
    mask_opened = binary_opening(mask_filled, selem)
    
    # Step 5: Closing (dilation + erosion) to connect nearby regions
    mask_closed = binary_closing(mask_opened, disk(5))
    
    return mask_closed.astype(np.uint8)

# Usage
cleaned_mask = morphological_postprocess(predicted_mask > 0.5)
```

### **Advanced: Adaptive Morphology**

```python
def adaptive_morphology(mask, object_size_percentile=0.1):
    """
    Adaptively determine morphological parameters based on mask statistics
    """
    # Calculate object sizes
    from skimage.measure import label, regionprops
    
    labeled = label(mask)
    props = regionprops(labeled)
    
    if len(props) > 0:
        areas = [prop.area for prop in props]
        min_size = int(np.percentile(areas, object_size_percentile * 100))
    else:
        min_size = 100
    
    # Apply morphology with adaptive parameters
    mask_clean = remove_small_objects(mask.astype(bool), min_size=min_size)
    mask_filled = binary_fill_holes(mask_clean)
    
    # Adaptive structuring element size
    radius = max(2, min_size // 100)
    mask_final = binary_closing(mask_filled, disk(radius))
    
    return mask_final.astype(np.uint8)
```

## 🔗 **4. CRF (CONDITIONAL RANDOM FIELD) POST-PROCESSING**

CRF refines segmentation by fixing small islands of misclassification, using both spatial information and RGB features.

### **Using pydensecrf Library:**

```python
# Install: pip install pydensecrf
import pydensecrf.densecrf as dcrf
from pydensecrf.utils import unary_from_softmax

def crf_refine(image, probabilities, n_iters=5, sxy_gaussian=3, compat_gaussian=3,
               sxy_bilateral=80, srgb_bilateral=13, compat_bilateral=10):
    """
    Apply CRF post-processing to refine segmentation
    
    Args:
        image: Original RGB image (H, W, 3), values 0-255
        probabilities: Model output probabilities (2, H, W) for [background, foreground]
        n_iters: Number of CRF iterations
    """
    h, w = probabilities.shape[1:]
    
    # Setup CRF
    d = dcrf.DenseCRF2D(w, h, 2)  # width, height, nlabels
    
    # Unary potential from network
    U = unary_from_softmax(probabilities)
    d.setUnaryEnergy(U)
    
    # Pairwise potential: spatial smoothness
    d.addPairwiseGaussian(sxy=sxy_gaussian, compat=compat_gaussian)
    
    # Pairwise potential: appearance (bilateral)
    d.addPairwiseBilateral(
        sxy=sxy_bilateral,
        srgb=srgb_bilateral,
        rgbim=image.astype(np.uint8),
        compat=compat_bilateral
    )
    
    # Inference
    Q = d.inference(n_iters)
    Q = np.array(Q).reshape((2, h, w))
    
    # Get refined mask
    refined_mask = np.argmax(Q, axis=0)
    
    return refined_mask
```
